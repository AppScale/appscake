229 def redirect_standard_io(timestamp)
 
230   begin
 
231     orig_stderr = $stderr.clone
 
232     orig_stdout = $stdout.clone
 
233     log_path = File.join(File.expand_path(File.dirname(__FILE__)), "..", "logs")
 
234     $stderr.reopen File.new(File.join(log_path, "deploy-#{timestamp}.log"), "w")
 
235     $stderr.sync = true
 
236     $stdout.reopen File.new(File.join(log_path, "deploy-#{timestamp}.log"), "w")
 
237     $stdout.sync = true

239   rescue Exception => e

241     $stdout.reopen orig_stdout
 
242     $stderr.reopen orig_stderr

244   ensure

247   end
